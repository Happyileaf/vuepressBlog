```javascript
function isPowerOfFour(n) {
    // write code here
    let a = 1
    let b = 4
    while (1) {
        if (n === a || n === b) {
            return true
        } else {
            if (a < n && n < b) {
                return false
            }
            a = b
            b = b * 4
        }
    }

}
```
